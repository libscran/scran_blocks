40 SingleQuantile(const Size_ n, const double quantile) {
41 const double frac = static_cast<double>(n - 1) * quantile;
42 const double base = std::floor(frac);
43 my_lower = static_cast<Size_>(base);
44 my_upper_fraction = frac - base;
45 my_lower_fraction = 1.0 - my_upper_fraction;
46 my_skip_upper = my_upper_fraction == 0;
47 }
48
49private:
50 Size_ my_lower;
51 double my_upper_fraction;
52 double my_lower_fraction;
53 bool my_skip_upper;
54
55public:
64 template<typename Iterator_>
65 Output_ operator()(Iterator_ begin, Iterator_ end) const {
66 auto target = begin + static_cast<typename std::iterator_traits<I<decltype(begin)> >::difference_type>(my_lower);
67 std::nth_element(begin, target, end);
68
69 // Avoid looking at target+1 if don't need it - in particular, we'd get a out-of-bounds access if quantile = 1.
70 if (my_skip_upper) {
71 return static_cast<Output_>(
72 *target
73 );
74 } else {
75 return static_cast<Output_>(
76 static_cast<double>(*target) * my_lower_fraction + static_cast<double>(*(target + 1)) * my_upper_fraction
77 );
78 }
79 }

100void parallel_quantiles(const std::size_t n, const std::vector<Stat_*>& in, const double quantile, Output_* const out, const bool skip_nan) {
101 const auto nblocks = in.size();
102 if (nblocks == 0) {
103 std::fill_n(out, n, std::numeric_limits<Output_>::quiet_NaN());
104 return;
105 } else if (nblocks == 1) {
106 std::copy_n(in[0], n, out);
107 return;
108 }
109
110 std::vector<Stat_> tmp_buffer;
111 tmp_buffer.reserve(nblocks);
112
113 if (skip_nan) {
114 for (std::size_t g = 0; g < n; ++g) {
115 tmp_buffer.clear();
116 for (I<decltype(nblocks)> b = 0; b < nblocks; ++b) {
117 const auto val = in[b][g];
118 if (!std::isnan(val)) {
119 tmp_buffer.push_back(val);
120 }
121 }
122
123 const auto actual_nblocks = tmp_buffer.size();
124 if (actual_nblocks == 0) {
125 out[g] = std::numeric_limits<Output_>::quiet_NaN();
126 } else if (actual_nblocks == 1) {
127 out[g] = tmp_buffer.front();
128 } else {
129 SingleQuantile<Output_, I<decltype(actual_nblocks)>> calc(actual_nblocks, quantile);
130 out[g] = calc(tmp_buffer.begin(), tmp_buffer.end());
131 }
132 }
133
134 } else {
135 SingleQuantile<Output_, I<decltype(nblocks)>> calc(nblocks, quantile);
136 for (std::size_t g = 0; g < n; ++g) {
137 tmp_buffer.clear();
138 for (I<decltype(nblocks)> b = 0; b < nblocks; ++b) {
139 tmp_buffer.push_back(in[b][g]);
140 }
141 out[g] = calc(tmp_buffer.begin(), tmp_buffer.end());
142 }
143 }
144}
