34template<typename Output_, class Iterator_>
35class SingleQuantile {
36public:
41 SingleQuantile(const std::size_t len, const double quantile) {
42 assert(len > 0);
43
44 if (quantile < 0 || quantile > 1) {
45 throw std::out_of_range("'quantile' should lie in [0, 1]");
46 }
47
48 const std::size_t len_m1 = len - 1;
49 const Output_ frac = static_cast<Output_>(len_m1) * static_cast<Output_>(quantile);
50 const Output_ base = std::floor(frac);
51
52 // Protect the cast from conversion imprecision between size_t and Float_,
53 // e.g., if it rounds up or if it converts it to an Inf.
54 my_lower = std::min(sanisizer::from_float<std::size_t>(base), len_m1);
55 my_upper_fraction = frac - base;
56 my_skip_upper = my_upper_fraction == 0;
57 }
58
59private:
60 std::size_t my_lower;
61 Output_ my_upper_fraction;
62 bool my_skip_upper;
63
64public:
74 Output_ operator()(Iterator_ begin, Iterator_ end) const {
75 assert(sanisizer::is_less_than(my_lower, end - begin));
76 auto target = begin + my_lower;
77 std::nth_element(begin, target, end);
78
79 // Avoid looking at target+1 if don't need it - in particular, we'd get a out-of-bounds access if quantile = 1.
80 const Output_ lower = *target;
81 if (my_skip_upper) {
82 return lower;
83 } else {
84 const Output_ upper = *std::min_element(target + 1, end);
85 return lower + (upper - lower) * my_upper_fraction;
86 }
87 }
88};

182template<typename Stat_, typename Output_>
183void parallel_quantiles(const std::size_t n, const std::vector<Stat_*>& in, const double quantile, Output_* const out, const bool skip_nan) {
184 const auto nblocks = in.size();
185 if (nblocks == 0) {
186 std::fill_n(out, n, std::numeric_limits<Output_>::quiet_NaN());
187 return;
188 } else if (nblocks == 1) {
189 std::copy_n(in[0], n, out);
190 return;
191 }
192
193 std::vector<Stat_> tmp_buffer;
194 tmp_buffer.reserve(nblocks);
195
196 if (skip_nan) {
197 SingleQuantileVariable<Output_, I<decltype(tmp_buffer.begin())> > calcs(nblocks, quantile);
198 for (std::size_t g = 0; g < n; ++g) {
199 tmp_buffer.clear();
200 for (I<decltype(nblocks)> b = 0; b < nblocks; ++b) {
201 const auto val = in[b][g];
202 if (!std::isnan(val)) {
203 tmp_buffer.push_back(val);
204 }
205 }
206 out[g] = calcs(tmp_buffer.size(), tmp_buffer.begin(), tmp_buffer.end());
207 }
208
209 } else {
210
211 SingleQuantile<Output_, I<decltype(tmp_buffer.begin())> > calc(nblocks, quantile);
212 for (std::size_t g = 0; g < n; ++g) {
213 tmp_buffer.clear();
214 for (I<decltype(nblocks)> b = 0; b < nblocks; ++b) {
215 tmp_buffer.push_back(in[b][g]);
216 }
217 out[g] = calc(tmp_buffer.begin(), tmp_buffer.end());
218 }
219 }
220}
