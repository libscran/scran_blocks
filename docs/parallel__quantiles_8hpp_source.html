39 SingleQuantile(const Size_ n, const double quantile) {
40 const double frac = static_cast<double>(n - 1) * quantile;
41 const double base = std::floor(frac);
42 my_lower = static_cast<Size_>(base);
43 my_upper_fraction = frac - base;
44 my_lower_fraction = 1.0 - my_upper_fraction;
45 my_skip_upper = my_upper_fraction == 0;
46 }
47
48private:
49 Size_ my_lower;
50 double my_upper_fraction;
51 double my_lower_fraction;
52 bool my_skip_upper;
53
54public:
63 template<typename Iterator_>
64 Output_ operator()(Iterator_ begin, Iterator_ end) const {
65 auto target = begin + static_cast<typename decltype(I(begin))::difference_type>(my_lower);
66 std::nth_element(begin, target, end);
67
68 // Avoid looking at target+1 if don't need it - in particular, we'd get a out-of-bounds access if quantile = 1.
69 if (my_skip_upper) {
70 return *target;
71 } else {
72 return *target * my_lower_fraction + *(target + 1) * my_upper_fraction;
73 }
74 }
75};

94template<typename Stat_, typename Output_>
95void parallel_quantiles(const std::size_t n, const std::vector<Stat_*>& in, const double quantile, Output_* const out, const bool skip_nan) {
96 const auto nblocks = in.size();
97 if (nblocks == 0) {
98 std::fill_n(out, n, std::numeric_limits<Output_>::quiet_NaN());
99 return;
100 } else if (nblocks == 1) {
101 std::copy_n(in[0], n, out);
102 return;
103 }
104
105 std::vector<Stat_> tmp_buffer;
106 tmp_buffer.reserve(nblocks);
107
108 if (skip_nan) {
109 for (std::size_t g = 0; g < n; ++g) {
110 tmp_buffer.clear();
111 for (decltype(I(nblocks)) b = 0; b < nblocks; ++b) {
112 const auto val = in[b][g];
113 if (!std::isnan(val)) {
114 tmp_buffer.push_back(val);
115 }
116 }
117
118 const auto actual_nblocks = tmp_buffer.size();
119 if (actual_nblocks == 0) {
120 out[g] = std::numeric_limits<Output_>::quiet_NaN();
121 } else if (actual_nblocks == 1) {
122 out[g] = tmp_buffer.front();
123 } else {
124 SingleQuantile<Output_, decltype(I(actual_nblocks))> calc(actual_nblocks, quantile);
125 out[g] = calc(tmp_buffer.begin(), tmp_buffer.end());
126 }
127 }
128
129 } else {
130 SingleQuantile<Output_, decltype(I(nblocks))> calc(nblocks, quantile);
131 for (std::size_t g = 0; g < n; ++g) {
132 tmp_buffer.clear();
133 for (decltype(I(nblocks)) b = 0; b < nblocks; ++b) {
134 tmp_buffer.push_back(in[b][g]);
135 }
136 out[g] = calc(tmp_buffer.begin(), tmp_buffer.end());
137 }
138 }
139}
