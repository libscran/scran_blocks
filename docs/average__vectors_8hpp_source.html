28template<bool weighted_, typename Stat_, typename Weight_, typename Output_>
29void compute(size_t n, std::vector<Stat_*> in, const Weight_* w, Output_* out, bool skip_nan) {
30 if (in.empty()) {
31 std::fill_n(out, n, std::numeric_limits<Output_>::quiet_NaN());
32 return;
33 } else if (in.size() == 1) {
34 if constexpr(weighted_) {
35 if (w[0] == 0) {
36 std::fill_n(out, n, std::numeric_limits<Output_>::quiet_NaN());
37 return;
38 }
39 }
40 std::copy(in[0], in[0] + n, out);
41 return;
42 }
43
44 std::fill_n(out, n, 0);
45 std::vector<Weight_> accumulated(skip_nan ? n : 0);
46
47 auto wcopy = w;
48 for (auto current : in) {
49 auto copy = out;
50
51 if constexpr(weighted_) {
52 // Don't skip if weight = 0, as we need to still compute the product, e.g.,
53 // if the value is Inf, we'd end up with 0 * Inf => NaN that can't be skipped.
54 Weight_ weight = *(wcopy++);
55
56 // Use the other loop and skip an unnecessary multiplication when the weight is 1.
57 if (weight != 1) {
58 if (skip_nan) {
59 for (size_t i = 0; i < n; ++i, ++current, ++copy) {
60 auto x = *current * weight;
61 if (!std::isnan(x)) {
62 *copy += x;
63 accumulated[i] += weight;
64 }
65 }
66 } else {
67 for (size_t i = 0; i < n; ++i, ++current, ++copy) {
68 *copy += *current * weight;
69 }
70 }
71 continue;
72 }
73 }
74
75 if (skip_nan) {
76 for (size_t i = 0; i < n; ++i, ++current, ++copy) {
77 auto x = *current;
78 if (!std::isnan(x)) {
79 *copy += x;
80 ++accumulated[i];
81 }
82 }
83 } else {
84 for (size_t i = 0; i < n; ++i, ++current, ++copy) {
85 *copy += *current;
86 }
87 }
88 }
89
90 if (skip_nan) {
91 for (size_t i = 0; i < n; ++i, ++out) {
92 *out /= accumulated[i];
93 }
94 } else {
95 double denom = 1;
96 if constexpr(weighted_) {
97 denom /= std::accumulate(w, w + in.size(), 0.0);
98 } else {
99 denom /= in.size();
100 }
101 for (size_t i = 0; i < n; ++i, ++out) {
102 *out *= denom;
103 }
104 }
105}
